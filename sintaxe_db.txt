CREATE DATABASE DADOS_CADASTRADOS;
CREATE TABLE CADASTRO_PROFESSOR
(
    CPF_MATRICULA VARCHAR(11) PRIMARY KEY,
    SENHA VARCHAR(12) NOT NULL UNIQUE,
    E_MAIL VARCHAR(255) NOT NULL UNIQUE,
    NOME VARCHAR(100) NOT NULL UNIQUE
);
CREATE TABLE CADASTRO_ALUNO
(
    CPF_MATRICULA VARCHAR(11) PRIMARY KEY,
    SENHA VARCHAR(12) NOT NULL UNIQUE,
    E_MAIL VARCHAR(255) NOT NULL UNIQUE,
    NOME VARCHAR(100) NOT NULL UNIQUE
);
