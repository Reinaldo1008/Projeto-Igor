CREATE TABLE IF NOT EXISTS USUARIOS (
    CPF_MATRICULA VARCHAR(11) PRIMARY KEY,
    SENHA VARCHAR(12) NOT NULL,
    E_MAIL VARCHAR(255) NOT NULL UNIQUE,
    NOME VARCHAR(100) NOT NULL UNIQUE,
    TIPO VARCHAR(10) NOT NULL
);

CREATE TABLE IF NOT EXISTS LIVROS (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    TITULO VARCHAR(200) NOT NULL,
    AUTOR VARCHAR(150) NOT NULL,
    CATEGORIA VARCHAR(100) NOT NULL,
    ISBN VARCHAR(20) UNIQUE,
    COPIAS INT NOT NULL DEFAULT 1,
    DISPONIVEIS INT NOT NULL DEFAULT 1
);

CREATE TABLE IF NOT EXISTS EMPRESTIMOS (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    CPF_ALUNO VARCHAR(11) NOT NULL,
    ID_LIVRO INT NOT NULL,
    DATA_SOLICITACAO DATE NOT NULL,
    DATA_DEVOLUCAO_PREVISTA DATE,
    DATA_DEVOLUCAO_REAL DATE,
    STATUS VARCHAR(20) NOT NULL DEFAULT 'pendente',
    FOREIGN KEY (CPF_ALUNO) REFERENCES USUARIOS(CPF_MATRICULA),
    FOREIGN KEY (ID_LIVRO) REFERENCES LIVROS(ID)
);

CREATE TABLE IF NOT EXISTS MENSAGENS (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    CPF_REMETENTE VARCHAR(11) NOT NULL,
    NOME_REMETENTE VARCHAR(100) NOT NULL,
    TIPO_REMETENTE VARCHAR(10) NOT NULL,
    MENSAGEM TEXT NOT NULL,
    RESPOSTA TEXT,
    DATA_ENVIO DATE NOT NULL
);
